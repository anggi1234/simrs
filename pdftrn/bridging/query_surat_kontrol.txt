SELECT
	PASIEN_DIAGNOSA.NO_REGISTRATION,
	PASIEN_DIAGNOSA.THENAME,
	PASIEN_DIAGNOSA.THEADDRESS,
	SEX.NAME_OF_GENDER,
	CLINIC.NAME_OF_CLINIC,
	CONCAT ( PASIEN_DIAGNOSA.AGEYEAR, 'Th. ', PASIEN_DIAGNOSA.AGEMONTH, 'Bln. ', PASIEN_DIAGNOSA.AGEDAY, 'Hr.' ) AS USIA,
	PASIEN_DIAGNOSA.DIAGNOSA_DESC,
	PASIEN_DIAGNOSA.TGLKONTROL,
	PASIEN_DIAGNOSA.DOCTOR,
	CONVERT(VARCHAR, GETDATE(), 106)PRINT_DATE
FROM
	PASIEN_DIAGNOSA
	LEFT JOIN SEX ON PASIEN_DIAGNOSA.GENDER = SEX.GENDER
	LEFT JOIN CLINIC ON PASIEN_DIAGNOSA.CLINIC_ID = CLINIC.CLINIC_ID 
WHERE
	PASIEN_DIAGNOSA.VISIT_ID = '202107290849050927792'
